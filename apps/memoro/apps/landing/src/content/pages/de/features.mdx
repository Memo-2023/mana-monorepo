{/* Hilfsfunktion zur Bereinigung des Slugs */}
export function cleanSlug(slug, lang) {
  const langPrefix = `${lang}/`;
  return slug.startsWith(langPrefix) ? slug.substring(langPrefix.length) : slug;
}
