import TeamCard from "../../../components/TeamCard.astro";
import FAQSection from "../../../components/FAQSection.astro";

export const TeamContent = ({ team, lang }) => {
  // Define categories
  const categories = frontmatter.sections.categories;

// Group team by categories
const teamByCategory = Object.keys(categories).reduce((acc, category) => {
acc[category] = team.filter((member) => member.data.category === category);
return acc;
}, {});

return (
<>
<div class="text-center max-w-3xl mx-auto mb-16">
<h1 class="text-4xl font-bold text-text-primary mb-4">
{frontmatter.sections.hero.title}
</h1>
<p class="text-xl text-text-secondary">
{frontmatter.sections.hero.subtitle}
</p>
</div>

      {Object.entries(teamByCategory).map(
        ([category, members]) =>
          members.length > 0 && (
            <div class="mb-16 last:mb-0 max-w-4xl mx-auto">
              <h2 class="text-2xl font-bold text-text-primary mb-8">
                {categories[category]}
              </h2>
              <div class="grid grid-cols-1 md:grid-cols-2 gap-8">
                {members.map((member) => (
                  <TeamCard
                    title={member.data.title}
                    role={member.data.role}
                    description={member.data.description}
                    image={member.data.image}
                    slug={member.data.slug || member.slug}
                    lang={lang}
                    social={member.data.social}
                  />
                ))}
              </div>
            </div>
          )
      )}

      <div class="bg-background-card rounded-xl p-8 mt-20 max-w-4xl mx-auto">
        <FAQSection
          title={frontmatter.sections.faq.title}
          categories={['team']}
        />
      </div>
    </>

);
};

<TeamContent {...props} />{" "}
